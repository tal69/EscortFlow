\begin{algorithm}[t]
\caption{\textsc{OneStep}$(L_x,L_y,O,A,E)$ in acyclic mode}
\label{alg: OneStep}
\KwIn{Grid dimensions $(L_x,L_y)$, output set $O$, target dictionary $A$, escort set $E$}
\KwOut{Updated target dictionary $A$, updated escort set $E$, and movement list $M$}
Remove from $A$ all entries whose key belongs to $O$\;
\If{$A = \emptyset$}{
    \textbf{return} $(A,E,\emptyset)$\;
}
$M \gets \emptyset$\tcp*{escort moves executed in the current time step}
$U \gets \emptyset$\tcp*{cells reserved against conflicting escort moves}
$H \gets \emptyset$\tcp*{target-load ids already moved indirectly}
Let $((a_1,\eta_1),\dots,(a_k,\eta_k))$ be the entries of $A$ sorted by increasing id\;
\For{$i \gets 1$ \KwTo $k$}{
    $a \gets a_i$; $\eta \gets \eta_i$\;
    \If{$\eta \in H$}{
            \textbf{continue}\;
    }
    Partition the escorts into $E_A(a),E_B(a),E_C(a),E_D(a)$ according to the zone definition\;
    \ForEach{escort $e \in E_A(a)$ in increasing distance order}{
        \If{$e$ can be moved to the location of $a$}{
            Move $e$ to advance $a$ by one cell toward its designated output\;
            Extend the same move, if possible, to promote lower-priority target loads lying farther ahead\;
            Update $A,E,M,U,H$\;
            Preserve the id of every moved target load at its new location\;
            \textbf{break}\;
        }
    }
    Add the current location of $a$ to $U$\tcp*{even if $a$ did not move}
    \ForEach{escort $e \in E_B(a)$ in increasing distance order}{
        \If{$e$ can be moved from Zone $B$ to Zone $A$}{
            Move $e$ from Zone $B$ to Zone $A$\;
            Extend the same move, if possible, to promote lower-priority target loads\;
            Update $A,E,M,U$\;
            \textbf{break}\;
        }
    }
    \ForEach{escort $e \in E_C(a)$ in increasing distance order}{
        \If{$e$ can be moved from Zone $C$ to Zone $B$}{
            Move $e$ from Zone $C$ to Zone $B$\;
            Extend the same move, if possible, to promote lower-priority target loads\;
            Update $A,E,M,U$\;
            \textbf{break}\;
        }
    }
    \ForEach{escort $e \in E_D(a)$ in increasing distance order}{
        \If{$e$ can be moved from Zone $D$ to Zone $C$}{
            Move $e$ from Zone $D$ to Zone $C$\;
            Extend the same move, if possible, to promote lower-priority target loads\;
            Update $A,E,M,U$\;
            \textbf{break}\;
        }
    }
}
\textbf{return} $(A,E,M)$\;
\end{algorithm}
